### Example 1: Initialize Integration

```python
from app.integrations.integration import initialize_integration

# Initialize on startup
await initialize_integration()
```

### Example 2: Call Analytics Service

```python
from app.integrations.integration import get_integration

integration = get_integration()

# Call Analytics for case prediction
prediction = await integration.call_service(
    service_id="itechsmart-analytics",
    endpoint="/api/v1/predict",
    method="POST",
    data={
        "case_type": "Personal Injury",
        "jurisdiction": "California",
        "damages_claimed": 500000,
        "evidence_strength": "strong"
    }
)

print(f"Predicted outcome: {prediction['outcome']}")
print(f"Confidence: {prediction['confidence']}%")
```

### Example 3: Report Error to Ninja

```python
# Report error for self-healing
try:
    process_document(doc_id)
except Exception as e:
    result = await integration.report_error(
        error=e,
        severity="high",
        context={"document_id": doc_id}
    )
    
    if result and result.get("auto_fix_available"):
        print(f"Ninja will fix: {result['fix_description']}")
```

### Example 4: Discover Services

```python
# Discover all available services
services = await integration.discover_services()

for service in services:
    print(f"{service['service_name']}: {service['capabilities']}")
```
